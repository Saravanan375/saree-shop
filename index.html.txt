<script>
const API_URL = "https://script.google.com/macros/s/AKfycbyi-0v150TEXg5a3i0vAg6KO6NMcBvIsPOfUa-jFYgSw3u6ItEphANbBkqzU_hLwMYlUQ/exec";  // <-- Replace this

fetch(API_URL + "?action=inventory")
  .then(res => res.json())
  .then(data => {
    if (!data.ok) {
      document.getElementById("productArea").innerHTML = "❌ Failed to load products";
      return;
    }
    renderProducts(data.inventory);
  })
  .catch(err => {
    console.log(err);
    document.getElementById("productArea").innerHTML = "❌ Error fetching product list";
  });

function renderProducts(list) {
  const box = document.getElementById("productArea");
  box.innerHTML = "";

  list.forEach(item => {
    box.innerHTML += `
      <div class="product-card">
        <img src="${item.image_url}" alt="${item.name}">
        <h3>${item.name}</h3>
        <p>₹${item.price}</p>
        <p>${item.description}</p>
        <button>Add to Cart</button>
      </div>
    `;
  });
}
</script>
